--ARCH--INSTALL
#INSTALLATION
sudo pacman -S docker
sudo systemctl enable docker.service
sudo systemctl start docker.service --now

#BASIC COMMANDS

sudo docker info

#PULL DEBIAN IMAGE
sudo docker pull debian

sudo docker images

#INIT/RUN DOCKER
sudo docker --name debian-docker -it -d debian

#SHOWS DOCKER STATUS
sudo docker ps -a

#STARTS IT - need to be done to be able to attach it
sudo docker start <insert its name>

#ATTACH DOCKER/"LOGS INTO DOCKER"
sudo docker attach debian-docker
exit (to leave docker)

#REMOVE DOCKER
sudo docker rm debian-docker
