-- INSTALLATION --
sudo pacman -S wireshark-qt wireshark-cli

-- SETUP --
sudo usermod -aG wireshark $USER
sudo modprobe usbmon

-- DOCKER SETUP --
sudo docker pull debian
sudo docker --name debian-docker -it -d debian
sudo docker run --name <docker_name> --privileged -v /dev/bus/usb:/dev/bus/usb -it -d debian

-- DOCKER START --
sudo docker start <docker_name>
sudo docker attach <docker_name>

-- REMOVE DOCKER --
sudo docker rm <docker_name>
sudo docker rm im <docker_img>

-- EDGETPU ---

sudo pip3 install Pillow
